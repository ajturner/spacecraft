31 ssfJulianDate JulianDate(const ssfSeconds &_time)
32 {
33  time_t t = static_cast<time_t>(floor(_time));
34  tm DateTime = *gmtime(&t);
35  ssfJulianDate JD = 0;
36  double uSec = _time - floor(_time);
37  JD += 367 * (DateTime.tm_year + 1900);
38  JD -= floor((7 * ((DateTime.tm_year + 1900) + floor(((DateTime.tm_mon + 1) + 9)/12))) / 4);
39  JD += floor((275 * (DateTime.tm_mon + 1)) / 9);
40  JD += DateTime.tm_mday;
41  JD += 1721013.5;
42  JD += (((DateTime.tm_sec + uSec)/60.0 + DateTime.tm_min) / 60.0 + DateTime.tm_hour) / 24;
43  return JD;
44 }
