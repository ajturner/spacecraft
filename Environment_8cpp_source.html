177 void EnvFunction::AddParameter(void* _parameter, int _paramNumber)
178 {
179  if (_paramNumber > m_NumParameters)
180  {
181  // Calculate the number of parameters "missing" to be temporarily filled
182  int delta = _paramNumber - m_NumParameters;
183  if(delta > 1)
184  {
185  for(int ii = 1; ii < delta; ++ii)
186  {
187  m_EnvFuncParameters.push_back(0);
188  }
189  }
190  m_EnvFuncParameters.push_back(_parameter);
191  }
192  else
193  m_EnvFuncParameters.insert(m_EnvFuncParameters.begin() + _paramNumber - 1, _parameter);
194 
195  m_NumParameters = _paramNumber;
196 
197 }
