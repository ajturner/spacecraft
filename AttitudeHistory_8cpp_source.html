119 void AttitudeHistory::GetState(const ssfTime& _requestedTime, Rotation& _returnRotation, Vector& _returnAngVelVector)
120 {
121  int RotVectorSize = QUATERNION_SIZE;
122  int AngVelVectorSize = (m_AttitudeHistory.back()).GetAngularVelocity().getIndexBound();
123 
124  // History::GetState returns the index of the nearest state.
125  vector<ssfTime>::difference_type nearestTimeIndex = History::GetState(_requestedTime);
126 
127  int numDataPoints = 0;
128  if(m_OriginalInterpolator)
129  numDataPoints = m_OriginalInterpolator->GetNumberDataPoints();
130  else
131  return;
132  Vector timeVector(numDataPoints);
133  Matrix dataMatrix(numDataPoints, RotVectorSize+AngVelVectorSize);
134  Vector tempVector(RotVectorSize+AngVelVectorSize);
135 
136  // See if there is data at the requested time.
137  if(m_TimeHistory[nearestTimeIndex] != _requestedTime)
138  {
139  // Determine if there is currently an interpolation through the requested point.
140  // if there isn't, create one, then evaluate the interpolation at the requested data point.
141  while(m_AttitudeInterpolations.capacity() < nearestTimeIndex)
142  {
143  m_AttitudeInterpolations.reserve(m_AttitudeHistory.capacity() + HISTORY_RESERVE_SIZE);
144  }
145  // Check to see if the data is initialized to hold an interpolation
146  if(!m_AttitudeInterpolations[nearestTimeIndex])
147  {
148  // fill in the 'missing' interpolations
149  for(int kk = m_AttitudeInterpolations.size(); kk < nearestTimeIndex; ++kk)
150  m_AttitudeInterpolations.push_back(m_OriginalInterpolator->NewPointer());
151 
152  m_AttitudeInterpolations.push_back(m_OriginalInterpolator->NewPointer());
153  }
154  // Calculate the interpolation parameters if necessary
155  if(!m_AttitudeInterpolations[nearestTimeIndex]->GetValid())
156  {
157  for(int ii = 1; ii <= numDataPoints; ++ii)
158  {
159  timeVector(ii) = m_TimeHistory[nearestTimeIndex - (numDataPoints/2-1) + ii].GetSeconds();
160  dataMatrix(ii,_(1,RotVectorSize)) = ~m_AttitudeHistory[nearestTimeIndex - (numDataPoints/2-1) + ii].GetRotation().GetRotation(Quaternion_Type);
161  dataMatrix(ii,_(RotVectorSize+1,RotVectorSize+AngVelVectorSize)) = ~m_AttitudeHistory[nearestTimeIndex - (numDataPoints/2-1) + ii].GetAngularVelocity();
162  }
163 
164  m_AttitudeInterpolations[nearestTimeIndex]->Interpolate(timeVector, dataMatrix);
165  }
166 
167  tempVector = m_AttitudeInterpolations[nearestTimeIndex]->Evaluate(_requestedTime.GetSeconds());
168  _returnRotation.Set(Quaternion(~tempVector(_(1,4))));
169  _returnAngVelVector(_) = tempVector(_(5,7));
170  }
171  else
172  { // there happens to be meshpoint data at the requested time, return it.
173  _returnRotation = m_AttitudeHistory[nearestTimeIndex].GetRotation();
174  _returnAngVelVector = m_AttitudeHistory[nearestTimeIndex].GetAngularVelocity();
175  }
176 
177  return;
178 }
