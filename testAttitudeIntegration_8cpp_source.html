59  // made static, which causes the memory to be allocated the first time the function is called

61  static Vector stateDot(7);
62  static Matrix bMOI(3,3);
63  static Matrix qtemp(4,3);
64  static Matrix Tmoment(3,1);
65  static Vector qIn(4);
66  static Vector wIn(3);
67 
68  // Retrieve the current integration states to a quaternion and angular velocity vector
69  qIn = _integratingState(_(VectorIndexBase,VectorIndexBase + 3));
70  wIn = _integratingState(_(VectorIndexBase + 4,VectorIndexBase + 6));
71  qIn /= norm2(qIn);
72 
73  // Calculate qDot
74  qtemp(_(VectorIndexBase+0,VectorIndexBase+2),_(VectorIndexBase+0,VectorIndexBase+2)) = skew(qIn(_(VectorIndexBase+0,VectorIndexBase+2))) + qIn(VectorIndexBase+3) * eye(3);
75  qtemp(VectorIndexBase+3, _(VectorIndexBase+0,VectorIndexBase+2)) = -(~qIn(_(VectorIndexBase+0,VectorIndexBase+2)));
76  qtemp(_,MatrixIndexBase) = 0.5 * qtemp * wIn;
77 
78  // Get the moments of inertia and calculate the omegaDot
79  bMOI = _parameters(_(MatrixIndexBase+0,MatrixIndexBase+2),_(MatrixIndexBase+0,MatrixIndexBase+2));
80  Tmoment = (bMOI.inverse() * (Tmoment - skew(wIn) * (bMOI * wIn)));
