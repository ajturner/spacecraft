391 static Vector AttituteDynamics(const ssfTime &_time, const Vector& _integratingState, Orbit *_Orbit, Attitude *_Attitude, const Matrix &_parameters, const Functor &_forceFunctorPtr)
392 {
393  static Vector stateDot(7);
394  static Matrix bMOI(3,3);
395  static Matrix qtemp(4,3);
396  static Vector Tmoment(3);
397  static Vector qIn(4);
398  static Vector wIn(3);
399 
400  qIn = _integratingState(_(VectorIndexBase,VectorIndexBase + 3));
401  wIn = _integratingState(_(VectorIndexBase + 4,VectorIndexBase + 6));
402  qIn /= norm2(qIn);
403 
404  qtemp(_(VectorIndexBase+0,VectorIndexBase+2),_) = skew(qIn(_(VectorIndexBase+0,VectorIndexBase+2))) + qIn(VectorIndexBase+3) * eye(3);
405  qtemp(VectorIndexBase+3, _) = -(~qIn(_(VectorIndexBase+0,VectorIndexBase+2)));
406 
407  bMOI = _parameters(_(MatrixIndexBase+0,MatrixIndexBase+2),_(MatrixIndexBase+0,MatrixIndexBase+2));
408  Tmoment = _forceFunctorPtr.Call(_time, _Orbit->GetStateObject(), _Attitude->GetStateObject());
409 
410  stateDot(_(VectorIndexBase,VectorIndexBase + 3)) = 0.5 * qtemp * wIn;
411  stateDot(_(VectorIndexBase + 4,VectorIndexBase + 6)) = (bMOI.inverse() * (Tmoment - skew(wIn) * (bMOI * wIn)));
412 
413  return stateDot;
414 }
