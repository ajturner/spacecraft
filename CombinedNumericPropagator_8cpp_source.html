48 Matrix CombinedNumericPropagator::PropagateOrbit(const vector<ssfTime> &_propTime, const Vector &_initConditions)
49 {
50  // Create temporary orbit & attitude that have the same types (copy of) the current state objects
51  // the actual state vectors will be replaced, but we need to make sure to have the same
52  // representation & frame types
53  static OrbitState tempOrbit = m_pOrbitObject->GetStateObject();
54  static vector<ssfTime> timeVector; timeVector.clear(); // clear since static
55  static vector<OrbitState> orbitVector; orbitVector.clear(); // clear since static
56 
59  m_OrbitStateMeshPoints.initialize(m_pOrbitIntegrator->Integrate(
60  _propTime,
61  m_pOrbitObject->GetDynamicsEq(),
62  _initConditions,
63  m_pOrbitObject,
64  m_pAttitudeObject,
65  m_pOrbitObject->GetParameters(),
66  m_pOrbitObject->GetEnvironmentForcesFunctor()
67  ));
69  // Convert the meshpoints to stored objects and append them to the appropriate history.
70  for (int ii = MatrixIndexBase; ii < m_OrbitStateMeshPoints[MatrixRowsIndex].getIndexBound() + MatrixIndexBase; ++ii)
71  {
72  (m_pOrbitObject->GetStateConversion())(m_OrbitStateMeshPoints(ii,_(MatrixIndexBase+1,m_OrbitStateMeshPoints[MatrixColsIndex].getIndexBound())), tempOrbit);
73  timeVector.push_back(ssfTime(m_OrbitStateMeshPoints(ii,MatrixIndexBase)));
74  orbitVector.push_back(tempOrbit);
75  }
76 
77  // Store the new histories to the orbit & attitude objects
78  if(m_pOrbitObject)
79  m_pOrbitObject->GetHistoryObject().AppendHistory(timeVector, orbitVector);
80 
81  // Set the stored Attitude & Orbit
83  m_pOrbitObject->SetStateObject(tempOrbit);
84 
85  return m_OrbitStateMeshPoints;
