428 void DirectionCosineMatrix::Set(const Vector& _EulerAxis, const Angle& _EulerAngle)
429 {
430  // Calculations done for each element to increase speed
431  double Phi = 1 - cos(_EulerAngle);
432  double e1 = _EulerAxis(VectorIndexBase+0);
433  double e2 = _EulerAxis(VectorIndexBase+1);
434  double e3 = _EulerAxis(VectorIndexBase+2);
435  double cosPhi = cos(_EulerAngle);
436  double sinPhi = sin(_EulerAngle);
437 
438  // Column 1
439  (*this)(MatrixIndexBase+0,MatrixIndexBase+0) = e1 * e1 * Phi + cosPhi;
440  (*this)(MatrixIndexBase+1,MatrixIndexBase+0) = e2 * e1 * Phi - e3 * sinPhi;
441  (*this)(MatrixIndexBase+2,MatrixIndexBase+0) = e3 * e1 * Phi + e2 * sinPhi;
442 
443  // Column 2
444  (*this)(MatrixIndexBase+0,MatrixIndexBase+1) = e1 * e2 * Phi + e3 * sinPhi;
445  (*this)(MatrixIndexBase+1,MatrixIndexBase+1) = e2 * e2 * Phi + cosPhi;
446  (*this)(MatrixIndexBase+2,MatrixIndexBase+1) = e3 * e2 * Phi - e1 * sinPhi;
447 
448  // Column 3
449  (*this)(MatrixIndexBase+0,MatrixIndexBase+2) = e1 * e3 * Phi - e2 * sinPhi;
450  (*this)(MatrixIndexBase+1,MatrixIndexBase+2) = e2 * e3 * Phi + e1 * sinPhi;
451  (*this)(MatrixIndexBase+2,MatrixIndexBase+2) = e3 * e3 * Phi + cosPhi;
452  Normalize();
453  return;

471 void DirectionCosineMatrix::Set(const ModifiedRodriguezParameters& _MRP)
472 {
473  // Calculations done for each element to increase speed
474  Matrix tempMatrix = (~(_MRP) * _MRP);
475  double Sigma = 1 - tempMatrix(MatrixIndexBase,MatrixIndexBase);
476  double sigma1 = _MRP(VectorIndexBase+0);
477  double sigma2 = _MRP(VectorIndexBase+1);
478  double sigma3 = _MRP(VectorIndexBase+2);
479 
480  // Column 1
481  (*this)(MatrixIndexBase+0,MatrixIndexBase+0) = 4*(sigma1*sigma1 - sigma2*sigma2 - sigma3*sigma3) + Sigma*Sigma;
482  (*this)(MatrixIndexBase+1,MatrixIndexBase+0) = 8*sigma2*sigma1 - 4*sigma3*Sigma;
483  (*this)(MatrixIndexBase+2,MatrixIndexBase+0) = 8*sigma3*sigma1 + 4*sigma2*Sigma;
484 
485  // Column 2
486  (*this)(MatrixIndexBase+0,MatrixIndexBase+1) = 8*sigma1*sigma2 + 4*sigma3*Sigma;
487  (*this)(MatrixIndexBase+1,MatrixIndexBase+1) = 4*(-sigma1*sigma1 + sigma2*sigma2 - sigma3*sigma3) + Sigma*Sigma;
488  (*this)(MatrixIndexBase+2,MatrixIndexBase+1) = 8*sigma3*sigma2 - 4*sigma1*Sigma;
489 
490  // Column 3
491  (*this)(MatrixIndexBase+0,MatrixIndexBase+2) = 8*sigma1*sigma3 - 4*sigma2 * Sigma;
492  (*this)(MatrixIndexBase+1,MatrixIndexBase+2) = 8*sigma2*sigma3 + 4*sigma1*Sigma;
493  (*this)(MatrixIndexBase+2,MatrixIndexBase+2) = 4*(-sigma1*sigma1 - sigma2*sigma2 + sigma3*sigma3) + Sigma*Sigma;
494 
495  // Multiplying factor
496  (*this) /= pow(1 + tempMatrix(MatrixIndexBase,MatrixIndexBase), 2);
497  Normalize();
498  return;
